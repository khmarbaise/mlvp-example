39  /**

48   * The following rules are read from the configuration file
49   * (svnaccess-groups-in-groups.conf).

64   * &#64;e-developer = rw
65   * [global:/project/trunk]
66   * &#64;all-developer = rw
67   * </pre>
68   *
69   * The tests will check if the AccessLevel will be extracted correctly and
70   * furthermore if the access to different directories inside this repository

205     @Test(dataProvider = "createAccessSet")
206     public void accessRuleCheck(String user, String repository,
207             String accessPath, AccessLevel expectedLevel) {
208         // Check the permission of the current user in the repository within the
209         // given accessPath
210         AccessLevel al_user = accessRules.getAccess(user, repository, accessPath);
211         Assert.assertEquals(expectedLevel, al_user);
