30  /**
31   * This test will check if the given users and group members
32   * do have the correct access permissions within the different
33   * folders and/or repositories.
34   *
35   * <pre>
36   * [groups]
37   * developer = harry, brian
38   * admin = micheal
39   * [/]
40   * * =
41   * [repository:/test/trunk]
42   * &#64;developer = r
43   * &#64;admin = rw
44   * </pre>
45   *
46   * The tests will check if the AccessLevel will be extracted correctly and
47   * furthermore if the access to different directories inside this repository
48   * will work well.
49   *
50   * @author Karl Heinz Marbaise
51   *
52   */
53  public class AccessRulesDerivingPermissionsTest {
54  
55      private AccessRules accessRules;
56      private AccessRule accessRule;
57      /**
58       * This will setup the following set of rules.
59       *
60       * <pre>
61       * [groups]
62       * developer = harry, brian
63       * admin = micheal
64       * [/]
65       * * =
66       * [repository:/test/trunk]
67       * &#64;developer = r
68       * &#64;admin = rw
69       * </pre>
70       */
71      @BeforeMethod
72      public void beforeMethod() {
73  
74          // &#64;developer = harry, brian
75          User userHarry = UserFactory.createInstance("harry");
76          User userBrian = UserFactory.createInstance("brian");
77          Group developerGroup = new Group("developer");
78          developerGroup.add(userHarry);
79          developerGroup.add(userBrian);
80  
81  
82          // &#64;admin = michael
83          Group adminGroup = new Group("admin");
84          User userMicheal = UserFactory.createInstance("michael");
85          adminGroup.add(userMicheal);
86  
87          // [/]
88          // * =
89          User userAsterik = UserFactory.createInstance("*");
90          AccessRule accessRuleRoot = new AccessRule("/");
91          accessRuleRoot.add(userAsterik, AccessLevel.NOTHING);
92  
93          // [repository:/test/trunk]
94          // &#64;developer = r
95          // &#64;admin = rw
96          accessRule = new AccessRule("repository", "/test/trunk");
97          accessRule.add(developerGroup, AccessLevel.READ);
98          accessRule.add(adminGroup, AccessLevel.READ_WRITE);
99  
100         accessRules = new AccessRules();
101         accessRules.add(accessRuleRoot);
102         accessRules.add(accessRule);
103     }
104 
105     @Test
106     public void accessTest() {
107         AccessLevel al_harry = accessRule.getAccessForPrincipal("harry");
108         Assert.assertEquals(AccessLevel.READ, al_harry);
109 
110         AccessLevel al_brian = accessRule.getAccessForPrincipal("brian");
111         Assert.assertEquals(AccessLevel.READ, al_brian);
112 
113         AccessLevel al_hugo = accessRule.getAccessForPrincipal("hugo");
114         Assert.assertEquals(AccessLevel.NOTHING, al_hugo);
115 
116         AccessLevel al_micheal = accessRule.getAccessForPrincipal("michael");
117         Assert.assertEquals(AccessLevel.READ_WRITE, al_micheal);
118     }
119

153     @Test(dataProvider = "createAccessSet")
154     public void accessRuleCheck(String user, String repository,
155             String accessPath, AccessLevel expectedLevel) {
156         // Check the permission of the current user in the repository within the
