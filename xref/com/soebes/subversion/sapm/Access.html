24  /**
25   * This represents the {@link AccessLevel access level} plus the user/group for which
26   * this permission should be defined.
27   * @author Karl Heinz Marbaise
28   *
29   */
30  public class Access {
31  
32      private IPrincipal principal;
33      private AccessLevel level;
34  
35      private boolean negativeRule;
36  
37      public Access() {
38          setPrincipal(null);
39          setLevel(AccessLevel.NOTHING);
40          setNegativeRule(false);
41      }
42  
43      /**
44       * Create a new instance with the given parameters.
45       * @param principal Either a {@link User} or a {@link Group}.
46       * @param level The level for the access.
47       */
48      public Access(IPrincipal principal, AccessLevel level) {
49          super();
50          setPrincipal(principal);
51          setLevel(level);
52          setNegativeRule(false);
53      }
54  
55      public Access(IPrincipal principal, AccessLevel level, boolean negativeRule) {
56          super();
57          setPrincipal(principal);
58          setLevel(level);
59          setNegativeRule(negativeRule);
60      }
61  
62      public IPrincipal getPrincipal() {
63          return principal;
64      }
65  
66      public void setPrincipal(IPrincipal principal) {
67          this.principal = principal;
68      }
69  
70      public AccessLevel getLevel() {
71          return level;
72      }
73  
74      public void setLevel(AccessLevel level) {
75          this.level = level;
76      }
77  
78      public void setNegativeRule(boolean negativeRule) {
79          this.negativeRule = negativeRule;
80      }
81  
82      public boolean isNegativeRule() {
83          return negativeRule;
84      }
85  }
